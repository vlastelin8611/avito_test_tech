# fastapi – веб-фреймворк для быстрого создания api и обработки http-запросов
fastapi

# uvicorn – сервер для запуска fastapi-приложения, умеющий обрабатывать асинхронные запросы
uvicorn

# sqlalchemy – библиотека для работы с базой данных через orm, позволяет описывать таблицы как классы
sqlalchemy

# pydantic – библиотека для валидации данных, используемая fastapi для проверки входящих запросов
pydantic

# PyJWT – библиотека для работы с jwt-токенами (кодирование и декодирование токенов для авторизации)
PyJWT

# pytest – тестовый фреймворк для написания и запуска автоматических тестов
pytest

httpx  # библиотека для работы с http-запросами, нужна fastapi для тестового клиента
